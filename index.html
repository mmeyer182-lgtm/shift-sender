<input type="text" id="shiftInput" placeholder="Enter shift (ex: 3/25 7a-7p)">

<button onclick="sendShift()">Send</button>


<script>

function sendShift() {

  const shift = document.getElementById("shiftInput").value;


  fetch("https://shiftbot-production-f32e.up.railway.app/post_shift", {

    method: "POST",

    headers: {

      "Content-Type": "application/json"

    },

    body: JSON.stringify({ text: shift })

  })

  .then(() => {

    alert("Shift sent!");

    document.getElementById("shiftInput").value = "";

  })

  .catch(() => {

    alert("Error sending shift.");

  });

}

</script>
